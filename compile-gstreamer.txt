# Install libtool
	wget http://ftp.gnu.org/pub/gnu/libtool/libtool-2.4.tar.gz
	tar xf libtool-2.4.tar.gz
	./configure
	make
	sudo make install	

# Compile and install liborc from source
	wget https://ftp.mpi-sb.mpg.de/pub/mirrors/linux/debian/pool/main/o/orc/orc_0.4.23.orig.tar.xz
	tar xf orc_0.4.23.orig.tar.xz
	rm orc_0.4.23.orig.tar.xz
	cd orc-0.4.23
	./configure --disable-gtk-doc --prefix=/gstreamer --exec-prefix=/gstreamer --datarootdir=/gstreamer
	make -j12
	sudo make install

# Compile and install openh264 from source
	git clone --recursive https://github.com/cisco/openh264.gitc
	cd openh264
	make
	sudo make install

# Compile and install gme from source
	git clone --recursive https://github.com/mcfiredrill/libgme.git
	cmake -G "Unix Makefiles"
	make
	sudo make install

# Compile and install sndio from source
	cvs -d anoncvs@moule.caoua.org:/sndio checkout -P sndio
	cd sndio
	./configure --prefix=/gstreamer
	make

# Compile and install gstreamer from source

	cd gstreamer

	make -j12
	sudo make install
